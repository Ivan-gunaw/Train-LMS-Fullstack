var angka = prompt ('masukan angka : ');
// pada saat kita menggunakan "prompt", seluruh data yang kita "Input" kedalam fungsi otomatis akan menjadi "String"


if (angka === 1) {
    alert('anda memasukan angka 1');
}else if (angka === 2){
    alert('anda memasukan angka 2');
}else if (angka === 3){
    alert('anda memasukan angka 3');
}else{
    alert('angka yang anda masukan salah!)
}
// maka dari itu pada persamaan di atas -> kita membandingkan variable "Angka" dengan "Integer"
// oleh karena itu, hasil yang akan muncul (di log) -> adalah 'angka yang anda masukan salah!'

NOTES penting !!!
> (==) | menandakan perbandingan 
> (===) | menandakan identitas -> membandingkan nilai sekaligus type data variable
